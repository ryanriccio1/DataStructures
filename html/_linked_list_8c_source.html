33 }
34 // make sure newPosition is in proper range (size_t dictates it cannot be negative)
35 uint8_t ableToInsert = newPosition <= list->length;
36 if (ableToInsert)
37 {
38 // Create a new node containing the new entry
39 Node *newNodePtr = (Node *)malloc(sizeof(Node));
40 setupNode(newNodePtr, newData, dataSize);
41
42 // if we are inserting at the head or tail
43 if (newPosition == 0 || newPosition == list->length)
44 {
45 // set head
46 if (newPosition == 0)
47 {
48 // link our new node to previous head
49 newNodePtr->next = list->head;
50
51 // as long as this is not our first item
52 // link the previous head to our new node
53 if (list->length > 0)
54 {
55 list->head->prev = newNodePtr;
56 }
57
58 // set the head to our new node
59 list->head = newNodePtr;
60 }
61 // set tail
62 if (newPosition == list->length)
63 {
64 // doubly linked tail
65 newNodePtr->prev = list->tail;
66 list->tail = newNodePtr;
67 }
68 }
69
70 // as long as we are not at the head of the list
71 if (newPosition > 0)
72 {
73 Node *prevPtr = NULL;
74 Node *nextPtr = NULL;
75
76 // if there is no previous node, do nothing
77 prevPtr = getNodeByIndex(list, newPosition - 1);
78 if (prevPtr != NULL)
79 { // link previous node with our new node
80 newNodePtr->next = prevPtr->next;
81 prevPtr->next = newNodePtr;
82 }
83
84 // if there is no next node, do nothing
85 nextPtr = getNodeByIndex(list, newPosition + 1);
86 if (nextPtr != NULL)
87 { // link our next node with our new node
88 newNodePtr->prev = nextPtr->prev;
89 nextPtr->prev = newNodePtr;

442Node *merge(LinkedList *list, Node *first, Node *second, MergeSortComparison compareFunc)
443{
444 // if we have reached a single side, return it to the sorter
445 if (!first)
446 return second;
447
448 if (!second)
449 return first;
450
451 // if first < second (with multiple methods of comparing)
452 if ((compareFunc == PointerCompare && first->data < second->data) ||
453 (compareFunc == ValueCompare && compare(list->dataType, first->data, second->data) == 1) ||
454 (compareFunc == SizeCompare && first->dataSize < second->dataSize))
455 {
456 // get the next node in the proper order recursively
457 first->next = merge(list, first->next, second, compareFunc);
458 // link to the proper next node (doubly)
459 first->next->prev = first;
460 first->prev = NULL;
461 return first;
462 }
463 else
464 {
465 // get the next node that is greater than first recursively
466 second->next = merge(list, first, second->next, compareFunc);
467 // link to proper node doubly
468 second->next->prev = second;
469 second->prev = NULL;
470 return second;
471 }

489Node *split(Node *head)
490{
491 Node *A = head;
492 Node *B = head;
493 // A gets incremented twice as fast as B
494 // As long as A->next->next is not null,
495 // we haven't reached the middle yet
496 while (A->next && A->next->next)
497 {
498 A = A->next->next;
499 B = B->next;
500 }
501 // Once we reach the middle, our new head is
502 // the next item in the chain
503 // (to prevent issues when there are odd amounts of data)
504 Node *temp = B->next;
505 B->next = NULL; // unlink the nodes
506 return temp;
507}
